let currentQ = 0;

let userAnswers = [];

let quizStarted = false;


function startQuiz() {

    currentQ = 0;

    userAnswers = new Array(quizQuestions.length).fill(null);

    quizStarted = true;

    document.getElementById('quiz-start-screen').style.display = 'none';

    document.getElementById('quiz-active').classList.add('active');

    document.getElementById('quiz-results').classList.remove('show');

    renderQuestion();

}


function renderQuestion() {

    const q = quizQuestions[currentQ];

    const progress = ((currentQ + 1) / quizQuestions.length) * 100;

    document.getElementById('quiz-progress').style.width = progress + '%';


    const isAnswered = userAnswers[currentQ] !== null;

    const selectedAnswer = userAnswers[currentQ];


    document.getElementById('quiz-question-area').innerHTML = `

        <div class="quiz-question-card">

            <div class="question-meta">

                <span class="question-num">Question ${currentQ + 1} of ${quizQuestions.length}</span>

                <span class="question-topic">📚 ${q.topic}</span>

            </div>

            <div class="question-text">${q.q}</div>

            <div class="options-grid">

                ${q.options.map((opt, i) => {

                    let cls = '';

                    if (isAnswered) {

                        if (i === q.answer) cls = 'correct';

                        else if (i === selectedAnswer && i !== q.answer) cls = 'wrong';

                        else if (i === selectedAnswer) cls = 'selected';

                    }

                    return `<button class="option-btn ${cls}" onclick="selectAnswer(${i})" ${isAnswered ? 'disabled' : ''}>

                        <span class="option-letter">${['A','B','C','D'][i]}</span>

                        <span>${opt}</span>

                    </button>`;

                }).join('')}

            </div>

            ${isAnswered ? `

                <div class="explanation-box show">

                    <div class="explanation-title">${selectedAnswer === q.answer ? '✅ Correct!' : '❌ Incorrect'}</div>

                    <div class="explanation-text"><strong>Explanation:</strong> ${q.explanation}</div>

                </div>

            ` : ''}

        </div>

    `;


    document.getElementById('prev-btn').style.display = currentQ > 0 ? 'block' : 'none';

    const nextBtn = document.getElementById('next-btn');

    nextBtn.disabled = !isAnswered;

    nextBtn.textContent = currentQ === quizQuestions.length - 1 ? 'See Results 🎯' : 'Next →';

}


function selectAnswer(idx) {

    if (userAnswers[currentQ] !== null) return;

    userAnswers[currentQ] = idx;

    renderQuestion();

}


function nextQuestion() {

    if (currentQ < quizQuestions.length - 1) {

        currentQ++;

        renderQuestion();

    } else {

        showResults();

    }

}


function prevQuestion() {

    if (currentQ > 0) { currentQ--; renderQuestion(); }

}


function showResults() {

    document.getElementById('quiz-active').classList.remove('active');

    const results = document.getElementById('quiz-results');

    results.classList.add('show');


    const correct = userAnswers.filter((a, i) => a === quizQuestions[i].answer).length;

    const pct = Math.round((correct / quizQuestions.length) * 100);


    let grade = '', msg = '', color = '';

    if (pct >= 87) { grade = '⭐ Excellent!'; msg = "Outstanding performance! You're fully exam-ready!"; color = 'var(--success)'; }

    else { grade = '💪 Keep Going!'; msg = "Don't give up! Re-read the modules and try again."; color = 'var(--danger)'; }


    const wrongAnswers = quizQuestions.filter((q, i) => userAnswers[i] !== q.answer);


    results.innerHTML = `

        <div style="text-align:center;padding:30px 0;">

            <div style="font-size:5rem;margin-bottom:20px;">${pct >= 87 ? '🏆' : pct >= 67 ? '🎯' : pct >= 47 ? '📚' : '💪'}</div>

            <div style="font-size:1.8rem;font-weight:800;margin-bottom:10px;">${grade}</div>

            <div style="color:var(--text-light);margin-bottom:30px;">${msg}</div>

            <div class="results-breakdown">

                <div class="result-stat"><div class="result-stat-num" style="color:var(--success)">✅ ${correct}</div><div class="result-stat-label">Correct</div></div>

                <div class="result-stat"><div class="result-stat-num" style="color:var(--danger)">❌ ${quizQuestions.length - correct}</div><div class="result-stat-label">Incorrect</div></div>

                <div class="result-stat"><div class="result-stat-num" style="color:var(--primary)">${pct}%</div><div class="result-stat-label">Score</div></div>

            </div>

            ${wrongAnswers.length > 0 ? `

                <div class="answers-review">

                    <h3 style="margin-bottom:15px;color:white;">📖 Review Incorrect Answers</h3>

                    ${wrongAnswers.map((q, i) => {

                        const qi = quizQuestions.indexOf(q);

                        return `<div class="review-item wrong-ans">

                            <div class="review-q">Q${qi+1}: ${q.q}</div>

                            <div class="review-your">Your answer: ${q.options[userAnswers[qi]]}</div>

                            <div class="review-correct">✅ Correct: ${q.options[q.answer]}</div>

                        </div>`;

                    }).join('')}

                </div>

            ` : '<div class="info-box green"><div class="info-box-title">🎉 Perfect Score!</div><p>You answered all questions correctly! You are fully prepared for your exam!</p></div>'}

            <div style="margin-top:30px;display:flex;gap:15px;justify-content:center;flex-wrap:wrap;">

                <button class="quiz-btn quiz-btn-primary" onclick="startQuiz()">🔄 Retake Quiz</button>

                <button class="quiz-btn quiz-btn-secondary" onclick="showView('modules')">📚 Review Modules</button>

            </div>

        </div>

    `;

    window.scrollTo({ top: 0, behavior: 'smooth' });

}

let allCards = [];

let currentCards = [];

let fcIndex = 0;

let isFlipped = false;


function initFlashcards() {

    setFCCategory('all');

}


function setFCCategory(cat) {

    document.getElementById('flashcard').classList.remove('flipped');

    isFlipped = false;

    if (cat === 'all') {

        allCards = Object.values(flashcards).flat();

    } else {

        allCards = flashcards[cat] || [];

    }

    currentCards = [...allCards];

    fcIndex = 0;

    updateCard();

}


function updateCard() {

    if (currentCards.length === 0) return;

    const card = currentCards[fcIndex];

    document.getElementById('fc-front').textContent = card.term;

    document.getElementById('fc-back').textContent = card.def;

    document.getElementById('fc-counter').textContent = `Card ${fcIndex + 1} of ${currentCards.length}`;

    document.getElementById('flashcard').classList.remove('flipped');

    isFlipped = false;

}


function flipCard() {

    const card = document.getElementById('flashcard');

    card.classList.toggle('flipped');

    isFlipped = !isFlipped;

}


function nextCard() {

    fcIndex = (fcIndex + 1) % currentCards.length;

    updateCard();

}


function prevCard() {

    fcIndex = (fcIndex - 1 + currentCards.length) % currentCards.length;

    updateCard();

}
